stage1需为3类输出ckpt（`03-14-01-11-56`）。

  --exp_name dicece_coarse \

  --epochs 200 \
  --batch_size 2 \
  --lr 3e-3 \
  --patch 160 160 160 \
  --val_patch 160 160 160 \
  --sw_batch_size 1 \
  --val_every 3 \
  --num_workers 10 \
  --prefetch_factor 4 \
  --amp \
  --loss dicece \
